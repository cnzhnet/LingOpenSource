SELECT 
	t1.TABLE_NAME AS tableName,
	t1.COLUMN_NAME AS paramName,
	CASE ISNULL(t1.COLUMN_COMMENT) WHEN TRUE THEN '' ELSE t1.COLUMN_COMMENT END paramDescription, 
	t1.DATA_TYPE AS dbType, 
	CASE t1.COLUMN_DEFAULT WHEN 'NULL' THEN NULL ELSE t1.COLUMN_DEFAULT END defaultValue,  
	CASE t1.IS_NULLABLE WHEN 'NO' THEN 0 ELSE 1 END allowNull, 
	CASE t1.COLUMN_KEY WHEN 'PRI' THEN 1 ELSE 0 END isPrimary,
	CASE t1.EXTRA WHEN 'auto_increment' THEN 1 ELSE 0 END isIdentity
FROM
	INFORMATION_SCHEMA.COLUMNS t1 
where 
	t1.TABLE_SCHEMA = ?dbName
